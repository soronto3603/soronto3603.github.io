# 백준 알고리즘 15650 풀이

본 풀이는 15649 풀이를 기반으로 하고 있습니다.

15649 문제와 다른점은 

> 고른 수열은 오름차순이어야 한다.

위의 조건이 추가 되었다.
```
#include<iostream>
#include<vector>

using namespace std;

int N,M;

int arr[10];
bool visit[10] = {false ,};

void dfs(int cnt){
    if(cnt == M){
        for(int i=0;i<M;i++){
            cout << arr[i];
            if( i == N-1){

            }else cout << " ";
        }
        cout << "\n";
    }else{
        for(int i=1;i<=N;i++){
            if( visit[i] == false && arr[cnt-1] < i ){
                visit[i] = true;
                arr[cnt] = i;
                dfs(cnt+1);
                visit[i] = false; 
            }
        }
    }
}

int main(){
    cin>>N;
    cin>>M;
   
    dfs(0);
}
```
